189 void move(const glm::vec3 & translation)

194
199 void rotate(const glm::mat4 & rotation)
200 {
201 target = location + rotation * (target - location);
202 }

212
217 glm::mat4 get_transform_matrix_inverse() const
218 {
219 return glm::lookAt(
220 glm::vec3(location[0], location[1], location[2]),
221 glm::vec3(target[0], target[1], target[2]),
222 glm::vec3(0.0f, 1.0f, 0.0f)
223 );
224 }
225
226 private:
230 void calculate_projection_matrix()
231 {
232 projection_matrix = glm::perspective(glm::radians(fov), ratio, near_z, far_z);
