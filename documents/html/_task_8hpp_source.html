47 class Task

51 static atomic<bool> is_stop;

54 public:
59 Task(function<void()> task_func) : task_func(task_func) {}
60
64 virtual ~Task() = default;
65
69 virtual void execute() = 0;
70
74 void stop_execution()
75 {
76 std::lock_guard<mutex> lock(mtx);
77 finish_execution = true;
78 cv.notify_all();
79 }
80
84 void stop()
85 {
86 std::lock_guard<mutex> lock(mtx);
87 is_stop = true;
88 cv.notify_all();
89 }
90
94 void resume()
95 {
96 std::lock_guard<mutex> lock(mtx);
97 is_stop = false;
98 cv.notify_all();
99 }
100
101 protected:
106 bool shouldStop()
107 {
108 return is_stop.load();
109 }
110
115 bool shouldFinish()
116 {
117 return finish_execution.load();
118 }
119
123 void wait_for_resume()
124 {
125 std::unique_lock<mutex> lock(mtx);
126 cv.wait(lock, [this] {return !shouldStop() || shouldFinish(); });
127 }
128
129 protected:
130 function<void()> task_func;
131 };
