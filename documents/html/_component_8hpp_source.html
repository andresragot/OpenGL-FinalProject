95 mat4 get_transform_matrix()
96 {
97 mat4 transform_matrix(1);
98 transform_matrix = glm::translate(transform_matrix, position);
99 transform_matrix = glm::scale(transform_matrix, scale);
100
101 glm::quat quaternion_rotation = glm::quat(glm::radians(rotation));
102 transform_matrix *= glm::mat4_cast(quaternion_rotation);
103
104 if (parent)
105 transform_matrix = parent->get_transform_matrix() * transform_matrix;
106
107 return transform_matrix;
108 }
